class While
{
   public static void main(String[] args) 
     {
       int n;
       System.out.print("Enter value for Condition \n");
       Scanner r = new Scanner(System.in);
       n = r.nextInt();
 
       while(n>=0)
         {
           System.out.println("Learn Coding");
         }
      }
}
